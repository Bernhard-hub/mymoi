## üöÄ QUICK START BEISPIELE

### Voice Bot Integration

```typescript
// In action-handlers.ts
import { creativeExamples } from './lib/creative-examples'
import { officeExamples } from './lib/office-examples'
import { collaborationExamples } from './lib/collaboration-examples'

async function handleUserRequest(transcript: string, userId: number, phone: string) {
  // Office
  if (transcript.includes('word') && transcript.includes('onedrive')) {
    return await officeExamples.createAIDocumentInOneDrive(userId, transcript)
  }
  
  // Creative
  if (transcript.includes('video')) {
    return await creativeExamples.handleVideoCreation(transcript, userId, phone)
  }
  
  // Collaboration
  if (transcript.includes('canva')) {
    return await collaborationExamples.handleCanvaDesign(
      transcript, userId, phone, 'instagram'
    )
  }
}
```
